Def QuartersWindow()
  Width 600 MiniWindow({
    Using "EXIT" Button(true)
    Scrolling(Vertical {

      Horizontal {
        Using "title" Label()
        Stretch {}
        Using "currentLuxText" Label()
        {
          Using "currentLuxury" Label(color = ColorId GREEN)
          If "currInsuffLux" Using "currentLuxury" Label(color = ColorId YELLOW) 
        }
      }

      Height 10 {}

      Position(MIDDLE_STRETCHED_X, Horizontal {
        Using "quarters_header" Label()
        Stretch {}
        Using "owner_header"  Label()
      })

      Height 6 {}

      Using "elems" List(VERTICAL,
        Height 30 {
          Using "callback" Focusable NewHighlight()
          Using "callback" ButtonDual()

            Width 120
            {
              Using "luxury" Label(color = ColorId GREEN) 
              If "insuffLuxury" Using "luxury" Label(color = ColorId YELLOW)

            Using "level" Label()

            }

          })
        }
      )
    })
  })
End

{

  FocusKeysVertical()
